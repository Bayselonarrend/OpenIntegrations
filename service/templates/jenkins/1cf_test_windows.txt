def runCommand(String command) {
    if (isUnix()) {
        sh(script: command)
    } else {
        bat(script: command)
    }
}

def vrunnerLogIndicatesFailure(String logContent) {
    return (logContent.contains('Ошибка HTTP при обращении к серверу') || logContent.contains('Не могу установить соединение'))
}

def echoVrunnerLog(String libraryName, String logFile) {
    if (fileExists(logFile)) {
        def logContent = readFile(logFile)
        echo "=== vrunner log (${libraryName}) ==="
        echo logContent
        echo "===================================="
        return logContent
    }
    echo "WARNING: vrunner log file not found: ${logFile}"
    return ''
}

def stopIbsrv() {
    // убиваем только ibsrv, слушающий нужный порт
    bat(script: "@echo off\r\nfor /f \"tokens=5\" %%%%a in ('netstat -aon ^| findstr :%7 ^| findstr LISTENING') do taskkill /F /PID %%%%a 2>nul", returnStatus: true)
}

def startIbsrv() {
    stopIbsrv()
    def ibData = "%5OpenIntegrationsIB_%3"
    bat(script: "@echo off\r\nstart \"\" /B \"\"%6\"\" --data=\"${ibData}\" --disable-direct --db-path=\"./\" -p %7")
    sleep(time: 10, unit: 'SECONDS')
}

def runLibraryTest(String libraryName, String configPath) {
    catchError(buildResult: 'FAILURE', stageResult: 'FAILURE') {
        script {
            def logFile = "./test_results/${libraryName}.vrunner.log"
            def codeFile = "./test_results/${libraryName}.code"
            def vrunnerCmd = "\"%2\" run enterprise --ibsrv --nocacheuse --ibconnection /F%5OpenIntegrationsIB_%3 --command \"RunUnitTests=${configPath}\""

            def exitCode = bat(
                script: "@echo off\r\nchcp 65001 >nul\r\n${vrunnerCmd} > \"${logFile}\" 2>&1",
                returnStatus: true
            )

            def logContent = echoVrunnerLog(libraryName, logFile)

            if (vrunnerLogIndicatesFailure(logContent)) {
                error "vrunner failed for ${libraryName}: see log above"
            }

            if (exitCode != 0) {
                error "vrunner exited with code ${exitCode} for ${libraryName}"
            }

            if (!fileExists(codeFile)) {
                try {
                    timeout(time: 5, unit: 'MINUTES') {
                        waitUntil(quiet: true, initialRecurrencePeriod: 3) {
                            fileExists(codeFile)
                        }
                    }
                } catch (Exception ignored) {
                    echoVrunnerLog(libraryName, logFile)
                    error "Test result file not found for ${libraryName} after vrunner completed"
                }
            }

            if (!fileExists(codeFile)) {
                echoVrunnerLog(libraryName, logFile)
                error "Test result file not found for ${libraryName}"
            }

            if (fileExists("./test_results/${libraryName}.report")) {
                def reportContent = readFile("./test_results/${libraryName}.report")
                echo "=== ${libraryName} report content ==="
                echo reportContent
                echo "=========================="
            } else {
                echo "WARNING: ${libraryName}.report file not found"
            }

            def testExitCode = readFile(codeFile).trim().replaceAll("[^0-9]", "")
            echo "Exit code for ${libraryName}: ${testExitCode}"
            if (testExitCode != "0") {
                error "Tests ${libraryName} failed with exit code: ${testExitCode}"
            }
        }
    }
}

pipeline {
	agent {
		label 'windows'
	}
	parameters {
%4
	}
	stages {

		stage('Sync add-ins') {
			steps {
				script {
					if (isUnix()) {
						sh 'bash src/addins/sync-addins.sh'
					} else {
						bat 'call src\\addins\\sync-addins.bat'
					}
				}
			}
		}

		stage('Prepare Test Results Directory') {
            steps {
                script {
                    runCommand '''
                        if exist ".\\test_results" (
                            rmdir /s /q ".\\test_results"
                        )
                        mkdir ".\\test_results"
                    '''
                }
            }
        }

		stage('Create ReportPortal launch'){
				steps{
					bat encoding: 'UTF-8', script:'chcp 65001 & oscript ./ci/os/internal/Classes/ReportPortalMethods.os start "1C:Enterprise"'
				}
		}

		stage('Start ibsrv') {
			steps {
				script {
					startIbsrv()
				}
			}
		}

%1

	}
	post{
		always{
			script {
				catchError() {   
				    runCommand('oscript ./ci/os/internal/Classes/ReportPortalMethods.os stop')
			    }
				catchError() {
				    stopIbsrv()
				}
			}
		}
	}
}
