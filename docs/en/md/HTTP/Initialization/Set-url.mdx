```bsl title="1C:Enterprise/OneScript code example"
    URL = "https://bin.openintegrations.dev";
    URL = URL + "/get";

    Result = OPI_HTTPRequests.NewRequest()
        .Initialize()
        .SetURL(URL) // <---
        .ProcessRequest("GET")
        .ReturnResponseAsJSONObject();
```
