# Return response filename 
 Returns the path to the response body file



<Tabs>
  <TabItem value="params" label="Parameters" default>

`Function ReturnResponseFilename(Val Forced = False, Val ExceptionOnError = False) Export`

  | Parameter | CLI option | Type | Required | Description |
  |-|-|-|-|-|
  | Forced | - | Boolean | &#x2716; | False > The processor object will be returned instead of the response if there were errors in it |
  | ExceptionOnError | - | Boolean | &#x2716; | Causes an exception with a log if there were errors during processing |

    URL = "https://bin.openintegrations.dev";

        .ProcessRequest("GET")
        .ReturnResponseFilename();
