    URL = "https://httpbin.org";
    URL = URL + "/get";

    ProxySettings = New InternetProxy;
    ProxySettings.Set("https", "proxy.com", 443, "user", "password", False);

    Result = OPI_HTTPRequests.NewRequest()
        .Initialize()
        .SetURL(URL)
        .SetProxy(ProxySettings) // <---
        .ProcessRequest("GET", False)
        .ReturnConnection();
