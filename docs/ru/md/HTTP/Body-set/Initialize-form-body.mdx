# Инициализировать Form тело 
 Инициализирует тело пустой x-www-form-urlencoded коллекцией для дальнейшего наполнения

  |-|-|-|-|-|
  | ТипКорня | - | Строка | &#x2716; | Тип корня: Соответствие, Структура |

```bsl title="Пример использования для 1С:Предприятие/OneScript"
    URL = "https://bin.openintegrations.dev";
    URL = URL + "/post";

    Результат = OPI_ЗапросыHTTP.НовыйЗапрос()
        .Инициализировать(URL)
        .ИнициализироватьFormТело() // <---
        .ВставитьПолеТела("Поле1", "10")
        .ВставитьПолеТела("Поле2", "Текст")
        .ОбработатьЗапрос("POST")
        .ВернутьОтветКакJSONКоллекцию();
```
