# Добавить значение
 Добавляет значение в массив Form или Json тела

  | Параметр | CLI опция | Тип | Обяз. | Назначение |

```bsl title="Пример использования для 1С:Предприятие/OneScript"
    URL = "https://bin.openintegrations.dev";
    URL = URL + "/post";

    МассивОжидаемый = Новый Массив;
    МассивОжидаемый.Добавить("A");
    МассивОжидаемый.Добавить("B");

    Результат = OPI_ЗапросыHTTP.НовыйЗапрос()
        .Инициализировать(URL)
        .ИнициализироватьJsonТело()
        .ВставитьПолеТела("Поле1", 10)
        .ВставитьПолеТела("Поле3", Новый Массив)
        .ДобавитьЗначение("Поле3", "A") // <---
        .ДобавитьЗначение("Поле3", "B")
        .ОбработатьЗапрос("POST")
        .ВернутьОтветКакJSONКоллекцию();
```
