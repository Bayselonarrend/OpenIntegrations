```bsl title="Пример использования для 1С:Предприятие/OneScript"
    // Сервер
    Порт          = 5555;
    ОбъектСервера = OPI_ZeroMQ.ОткрытьПортRep(Порт);

    Если Не OPI_ZeroMQ.ЭтоОбъектКоннектора(ОбъектСервера) Тогда
        ВызватьИсключение OPI_Инструменты.JSONСтрокой(ОбъектСервера);
    КонецЕсли;

    // Клиент
    Адрес     = "tcp://127.0.0.1:5555";
    Результат = OPI_ZeroMQ.СоздатьСоединениеReq(Адрес);

    Если Не OPI_ZeroMQ.ЭтоОбъектКоннектора(Результат) Тогда
        ВызватьИсключение OPI_Инструменты.JSONСтрокой(Результат);
    КонецЕсли;

    Сообщение = СтрШаблон("ZMQ_C_REQ_%1", Формат(ТекущаяДата(), "ДФ=yyyyMMddhhmmss"));
    Данные = ПолучитьДвоичныеДанныеИзСтроки(Сообщение);

    Отправка = OPI_ZeroMQ.ОтправитьДанные(Результат, Данные, 3000);
    Проверка = OPI_ZeroMQ.ПолучитьДанные(ОбъектСервера, 3000);
```
