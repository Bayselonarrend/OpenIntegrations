    Мета.Добавить("tag");
    Мета.Добавить(1);
    Мета.Добавить(Ложь);
    Мета.Добавить(ПолучитьДвоичныеДанныеИзHexСтроки("0A0B"));
    Исходное.Вставить("items", Мета);
